
རྒྱུན་རིང་པོར་རྩེ་རྒྱུ་ཡོད་མ་རེད། ཆུ་ཚོད་གཉིས་གསུམ། ཆུ་ཚོད་གཅིག་གཉིས་ཞིག་རྩེ་ན་ཏག་ཏག་ཡོད་རེད། དངོས་ནས། ཁྱེད་རང་གི་ཁ་ལག་བཟོས། མ་བཟོས། ཁ་པར་བཟུང་། གློག་ཡོད། མྱིག་ནང་གི་མྱིག་སྐྱག་ལྟོས་དང་སོང་ངོ་བཀྲུས། བཞག་མི་དགོས་སོང་། སོང་ངོ་བཀྲུས་ཤོག་སོང་། ཨ་་་ང་ཡར་ར་ཡོང་གི་ཡིན། མི་དགོས་ཡ། སོང་།  #00:01:16-9#

རྒྱུག་ནས་མགྱོགས་ཙི་བཀྲུས་ཤོག མྱིག་སྐྱག་དག་མེད་གི དེ་མོ་མྱིག་རྫོབ་ཀྱི་མེད་ན་ཀླ་ས་ཀློའུ་སེ་ཞིག་ཡོད་གི་ར། མདང་ཅི་ཟེར་ནི་ཤེས། ངས་སྔུན་ན་རྒྱུགས་ཆེད་ཡོང་། རྒན་ལགས། མོག་མོག་ཟ་བསམ་ཀྱི་ཟེར། བསམ་ཀྱི་ཟེར་ནས། ཁོས་འཁུར་རྒྱུགས་ཡོང་། ཆོག་ནི་རེད་ཟེར། ཆོག་ནི་རེད་ཟེར་ནས། སྡེར་མ་འཁུར་ཅན་པོ་རེད། སོང་ནས་མོག་མོག་བཅུ་ཐམ་པ་འཁུར་རྒྱུགས་ཡོང་། ད་ནང་ཞིག་ལ་རྒན་ལགས་ཁྱེད་ཟོས་ཐ་ཞིམ་གྱི་ཟེར། མ་ཟོས་དེ་འཕྲོས་ཟ་དགོས། དེ་ནས་འཁྱག་སྒམ་ནང་རྫུས་ཞོག ཨ་ལ་དམྱལ་བ་པོ་དེའི་ཅིག་འཁྲབ་བཞག་ཐ། རྒན་ཁྱེད་game་དེ་རྩེ་ཞེས་ནི་མ་རེད། ཁོང་ཚོས་བསླབ་ཁྱེད་ལྟོས་ཟེར།
